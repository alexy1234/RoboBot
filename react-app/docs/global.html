Example usage of usePostData

function PostTest() {
  // Call usePostData custom hook
  const {res, loading, error} = usePostData("/post_example", {"name": "KEVIN"});

  return (
    <>
      {res && <p>{res}</p>}
      {loading && <p>{loading}</p>}
      {error && <p>{error}</p>}
    </>
  )
}
